50   class TrendingArticleSlider extends StatefulWidget {
51     @override
52     _TrendingArticleSliderState createState() => _TrendingArticleSliderState();
53   }
54   
55   class _TrendingArticleSliderState extends State<TrendingArticleSlider> {
56     int _current = 0;
57     final CarouselController _controller = CarouselController();
58   
59     @override
60     Widget build(BuildContext context) {
61       HomeProvider home = context.read<HomeProvider>();
62   
63       return Container(
64         width: MediaQuery.of(context).size.width,
65         color: HexColor("#E1fff7"),
66         margin: EdgeInsets.fromLTRB(10, 0, 10, 0),
67         child: Column(children: [
68           Container(
69             child: Row(
70               mainAxisSize: MainAxisSize.min,
71               children: [
72                 Expanded(
73                   child: Text(
74                     'TRENDING ARTICLES',
75                     style: TextStyle(
76                         fontSize: 18,
77                         fontWeight: FontWeight.bold,
78                         color: HexColor("#00AF8A")),
79                   ),
80                 ),
81                 Container(
82                   height: 35,
83                   child: Row(
84                     children: [
85                       Text(
86                         'See All',
87                         style: TextStyle(fontSize: 16, color: HexColor("#00AF8A"),),
88                       ),
89                       Icon(Icons.arrow_forward, size: 15, color: HexColor("#00AF8A"),),
90                     ],
91                   ),
92                 ),
93               ],
94             ),
95           ),
96           Container(
97             child: CarouselSlider(
98               carouselController: _controller,
99               options: CarouselOptions(
100                  initialPage: 0,
101                  aspectRatio: 2.5,
102                  disableCenter: true,
103                  viewportFraction: 0.5,
104                  scrollDirection: Axis.horizontal,
105                  onPageChanged: (index, reason) {
106                    setState(() {
107                      _current = index;
108                    });
109                  }),
110              items: List.generate(home.health.articles.length,
111                  (index) => home.health.articles[index].renderCard(context)),
112            ),
113          ),
114          Row(
115            mainAxisAlignment: MainAxisAlignment.center,
116            children: List.generate(home.health.articles.length, (index) {
117              return GestureDetector(
118                  onTap: () => _controller.animateToPage(index),
119                  child: Container(
120                    width: 8.0,
121                    height: 8.0,
122                    margin: EdgeInsets.symmetric(vertical: 8.0, horizontal: 4.0),
123                    decoration: BoxDecoration(
124                        shape: BoxShape.circle,
125                        color: (Theme.of(context).brightness == Brightness.dark
126                                ? Colors.grey
127                                : Color(0xFF00AF8A))
128                            .withOpacity(_current == index ? 0.9 : 0.4)),
129                  ));
130            }),
131          ),
132        ]),
133      );
134    }
